# 向量

相当于一个动态数组/可变长数组

```c++
#include<vector>
std::vector<Typename T>v;
//
std::vector<int>v;
int len=v.size();
int x;v.push_back(x);
v.pop_back();
for(int i=v.begin();i<=v.end();i++);
int x; v.insert(i,x);
int i; v.erase(i);
v.clear();
for(int i=0;i<v.size();i++)std::cout<<v[i];
```
